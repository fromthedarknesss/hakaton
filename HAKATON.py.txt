    },
    'Робітник 2': {
        'прізвище': 'Василенко',
        'посада': 'Програміст',
        'коефіцієнт ефективності': '79',

choise = input('Вітаємо! Виберіть будь-ласка цифру. 1 - вивести на екран прізвища всіх співробітників, 2 - вивести на екран прізвище самого ефективного співробітника, 3 - вивести на екран посади усіх співробітників, 4 - вивести всю інформацію')
if choise == '1':
    print('Прізвища всіх співробітників:')

        print('-',workers[worker]['прізвище'])

    for worker in workers:
        print('-',workers['Робітник 3']['прізвище'])

    for worker in workers:
